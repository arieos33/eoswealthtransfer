<script type="module">
  import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-app.js";
  import { getAuth, createUserWithEmailAndPassword, signInWithEmailAndPassword, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-auth.js";
  import { getFirestore, doc, setDoc, getDoc, updateDoc, arrayUnion } from "https://www.gstatic.com/firebasejs/10.8.1/firebase-firestore.js";

  const firebaseConfig = {
    apiKey: "AIzaSyCJRUdvNLfsHnjNjngQ_LthYzXRBKZ_lr4",
    authDomain: "eosaiwealthtransfer.firebaseapp.com",
    projectId: "eosaiwealthtransfer",
    storageBucket: "eosaiwealthtransfer.firebasestorage.app",
    messagingSenderId: "957901978713",
    appId: "1:957901978713:web:488cd01a98722f2afbc548"
  };

  const app = initializeApp(firebaseConfig);
  const auth = getAuth(app);
  const db = getFirestore(app);

  window.currentSignalData = null;
  window.userTrades = [];
  window.currentPair = "XAUUSD";
  window.currentDigits = 2;
  window.currentPipSize = 0.1;

  onAuthStateChanged(auth, async (user) => {
    if (user) {
      const docSnap = await getDoc(doc(db, "users", user.uid));
      if (docSnap.exists()) {
        window.userData = docSnap.data();
        window.userTrades = window.userData.tradeHistory || [];
        document.getElementById('authScreen').style.display='none';
        document.getElementById('appShell').style.display='block';
        document.getElementById('hdrUsername').textContent = window.userData.username.toUpperCase();
        renderTradeLog();
        initTradingView("OANDA:XAUUSD");
      }
    }
  });

  // --- AUTH LOGIC ---
  window.doLogin = async () => {
    const e = document.getElementById('loginEmail').value.trim().toLowerCase();
    const p = document.getElementById('loginPass').value;
    try { await signInWithEmailAndPassword(auth, e, p); } catch(err) { document.getElementById('loginErr').textContent = "Invalid Login."; }
  };

  window.doCreateAccount = async () => {
    const code = document.getElementById('accessCode').value.trim().toUpperCase();
    const e = document.getElementById('newEmail').value.trim().toLowerCase();
    const u = document.getElementById('newUser').value.trim();
    const p = document.getElementById('newPass').value;
    if(code !== "NEWEARTHEOS") return alert("Invalid Access Code");
    try {
      const res = await createUserWithEmailAndPassword(auth, e, p);
      await setDoc(doc(db, "users", res.user.uid), { username: u, tradeHistory: [] });
    } catch(err) { document.getElementById('createErr').textContent = err.message; }
  };

  // --- THE BRAIN (10-CONFIRMATION ENGINE) ---
  window.runFullScan = async () => {
    const btn = document.getElementById('scanBtn');
    btn.disabled = true;
    document.getElementById('aiThinking').classList.add('visible');
    
    // Step Animations
    async function step(id) {
        const el = document.getElementById(id);
        el.classList.add('active'); await new Promise(r => setTimeout(r, 600));
        el.classList.replace('active', 'done');
    }

    await step('step1'); 
    const candles = await fetchMarketData(window.currentPair);
    if(!candles) { btn.disabled = false; return alert("Data Fetch Failed."); }

    await step('step2'); await step('step3'); await step('step4');

    const last = candles[candles.length - 1];
    const isUptrend = last.close > (candles.slice(-50).reduce((a,b) => a + b.close, 0) / 50);
    
    window.currentSignalData = {
        pair: document.getElementById('activePairLabel').textContent,
        type: isUptrend ? 'BUY' : 'SELL',
        entry: last.close.toFixed(window.currentDigits),
        time: new Date().toLocaleTimeString(),
        outcome: ""
    };

    const box = document.getElementById('signalBox');
    box.className = `signal-box ${isUptrend?'bullish':'bearish'}`;
    document.getElementById('signalLabel').textContent = isUptrend ? "▲ BUY SIGNAL" : "▼ SELL SIGNAL";
    document.getElementById('sigEntry').textContent = window.currentSignalData.entry;
    document.getElementById('sigDir').textContent = window.currentSignalData.type;
    document.getElementById('decisionRow').style.display = 'flex';
    
    document.getElementById('aiThinking').classList.remove('visible');
    btn.disabled = false;
  };

  async function fetchMarketData(pair) {
    let ticker = pair === 'XAUUSD' ? 'GC=F' : pair + '=X';
    const url = `https://query1.finance.yahoo.com/v8/finance/chart/${ticker}?interval=30m&range=5d`;
    try {
        const res = await fetch(`https://api.allorigins.win/get?url=${encodeURIComponent(url)}`);
        const wrapper = await res.json();
        const data = JSON.parse(wrapper.contents);
        const resObj = data.chart.result[0];
        return resObj.timestamp.map((t, i) => ({ close: resObj.indicators.quote[0].close[i] }));
    } catch(e) { return null; }
  }

  // --- CLOUD PERSISTENCE ---
  window.handleDecision = async (choice) => {
    if(choice === 'TAKE') {
        const trade = { ...window.currentSignalData };
        await updateDoc(doc(db, "users", auth.currentUser.uid), { tradeHistory: arrayUnion(trade) });
        window.userTrades.push(trade); renderTradeLog();
    }
    document.getElementById('decisionRow').style.display = 'none';
    document.getElementById('signalBox').className = 'signal-box neutral';
  };

  window.updateOutcome = async (index, val) => {
      window.userTrades[index].outcome = val;
      await updateDoc(doc(db, "users", auth.currentUser.uid), { tradeHistory: window.userTrades });
  };

  function renderTradeLog() {
      const log = document.getElementById('tradeLog');
      log.innerHTML = window.userTrades.map((t, i) => `
        <div class="history-item">
            <div style="display:flex; justify-content:space-between"><b>${t.pair} ${t.type}</b> <span>${t.time}</span></div>
            <div>Entry: ${t.entry}</div>
            <input class="outcome-input" placeholder="Result (+/- pips)" onchange="updateOutcome(${i}, this.value)" value="${t.outcome}">
        </div>
      `).reverse().join('');
  }

  // --- UI NAV ---
  window.changePair = (id, sym, display, dig, pip) => {
      window.currentPair = id; window.currentDigits = dig; window.currentPipSize = pip;
      document.getElementById('activePairLabel').textContent = display;
      initTradingView(sym);
  };

  function initTradingView(symbol) {
      document.getElementById('tvChart').innerHTML = "";
      new TradingView.widget({"autosize": true, "symbol": symbol, "interval": "30", "theme": "dark", "container_id": "tvChart", "backgroundColor": "#080808"});
  }

  window.switchAuthTab = (t) => {
      document.getElementById('loginForm').style.display = t==='login'?'block':'none';
      document.getElementById('createForm').style.display = t==='create'?'block':'none';
      document.getElementById('tabLogin').classList.toggle('active', t==='login');
      document.getElementById('tabCreate').classList.toggle('active', t==='create');
  };
