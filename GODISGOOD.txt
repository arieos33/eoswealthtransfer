        document.getElementById('sigDir').textContent = window.currentSignalData.type;

        document.getElementById('sigSL').textContent = window.currentSignalData.sl;
        document.getElementById('sigTP').textContent = window.currentSignalData.tp;
        document.getElementById('sigType').textContent = window.currentSignalData.exec;
        document.getElementById('sigRR').textContent = `1:${(tp_pips/baseSL).toFixed(1)}`;
        document.getElementById('decisionRow').style.display = 'flex';
    } else {
        // STRICT CLEARING IF NO SETUP
        window.currentSignalData = null;
        box.className = 'signal-box neutral';
        document.getElementById('signalLabel').textContent = "NO VALID SETUP DETECTED";
        ['sigEntry','sigDir','sigSL','sigTP','sigType','sigRR'].forEach(id => {
            document.getElementById(id).textContent='—';
            document.getElementById(id).style.color='';
        });
        document.getElementById('decisionRow').style.display = 'none';
    }

    document.getElementById('checksList').style.display = 'block';
    document.getElementById('checksContainer').innerHTML = checks.map(c => `
        <div class="check-row">
            <span class="check-icon ${c.p?'check-pass':'check-fail'}">${c.p?'✓':'✕'}</span>
            <span style="color: ${c.p ? '#e8d8b0' : 'var(--muted)'}">${c.n}</span>
        </div>
    `).join('');

    status.textContent = "SCAN COMPLETE."; btn.disabled = false;
  };

  // --- DB & LOGIC ---
  window.handleDecision = async (choice) => {
    if(choice === 'TAKE' && window.currentSignalData) {
        const trade = { ...window.currentSignalData };
        await updateDoc(doc(db, "users", auth.currentUser.uid), { tradeHistory: arrayUnion(trade) });
        window.userTrades.push(trade); 
        renderTradeLog();
    }
    // Reset UI after decision
    document.getElementById('decisionRow').style.display = 'none';
    document.getElementById('checksList').style.display = 'none';
    document.getElementById('signalBox').className = 'signal-box neutral';
    document.getElementById('signalLabel').textContent = 'AWAITING NEXT SCAN';

        document.getElementById(id).textContent='—';
        document.getElementById(id).style.color='';
    });
  };

  function renderTradeLog() {
      let winCount = 0;
      const log = document.getElementById('tradeLog');
      
      if(!window.userTrades || window.userTrades.length === 0) {
          log.innerHTML = "<div style='color:var(--muted);font-size:11px;padding:8px;text-align:center'>No trades synced yet</div>";
          document.getElementById('statTrades').textContent = "0";
          document.getElementById('statWinRate').textContent = "--%";
          return;
      }

      log.innerHTML = window.userTrades.map((t, i) => {
          const isWin = t.outcome && (t.outcome.includes('+') || t.outcome.toLowerCase().includes('win'));
          if(isWin) winCount++;

          return `
            <div class="history-item ${isWin ? 'win' : ''}">
                <div style="display:flex; justify-content:space-between"><b>${t.pair} <span style="font-size:10px; color:${t.type==='BUY'?'var(--green)':'var(--red)'}">${t.type}</span></b> <span>${t.time}</span></div>
                <div style="font-size:10px; color:var(--text); margin-top:4px;">E: ${t.entry} | SL: ${t.sl} | TP: ${t.tp}</div>
                <div style="font-size:9px; color:var(--gold-dim); margin-top:2px;">${t.exec}</div>
                <input class="outcome-input" placeholder="Type outcome (+ pips or win) to save" onchange="updateOutcome(${i}, this.value)" value="${t.outcome}">
            </div>
          `;
      }).reverse().join('');

      // Update Stats
      document.getElementById('statTrades').textContent = window.userTrades.length;
      const totalFinished = window.userTrades.filter(t => t.outcome && t.outcome.trim() !== "").length;
      if(totalFinished > 0) {
          document.getElementById('statWinRate').textContent = Math.round((winCount/totalFinished)*100) + '%';
      } else {
          document.getElementById('statWinRate').textContent = '--%';
      }

      // 20-Win Divine Milestone
      if(winCount >= 20 && !window.milestone20Shown) {
          window.milestone20Shown = true;
          document.getElementById('divineModal').classList.add('open');
      }
  }
